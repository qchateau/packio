49 class incremental_parser {
50 public:
51  incremental_parser()

57  std::optional<request> get_request()
58  {
59  if (parsed_.empty()) {
60  return std::nullopt;
61  }
62  auto object = std::move(parsed_.front());
63  parsed_.pop();
64  return parse_request(std::move(object.as_object()));
65  }

73  parsed_.pop();

75  }

87  void buffer_consumed(std::size_t bytes)
88  { //
89  std::size_t parsed = 0;
90  while (parsed < bytes) {
91  parsed += parser_->write_some(
92  buffer_.data() + parsed, bytes - parsed);
93  if (parser_->done()) {
94  parsed_.push(parser_->release());
95  parser_->reset();
96  }
97  }
98  }

103  }

105 private:
106  static std::optional<response> parse_response(boost::json::object&& res)
107  {
108  auto id_it = res.find("id");
109  auto result_it = res.find("result");
110  auto error_it = res.find("error");
111 
112  if (id_it == res.end()) {
113  PACKIO_ERROR("missing id field");
114  return std::nullopt;
115  }
116  if (result_it == res.end() && error_it == res.end()) {
117  PACKIO_ERROR("missing error and result field");
118  return std::nullopt;
119  }
120 
121  std::optional<response> parsed{std::in_place};
122  parsed->id = std::move(id_it->value());
123  if (error_it != res.end()) {
124  parsed->error = std::move(error_it->value());
125  }
126  if (result_it != res.end()) {
127  parsed->result = std::move(result_it->value());
128  }
129  return parsed;
130  }
131 
132  static std::optional<request> parse_request(boost::json::object&& req)
133  {
134  auto id_it = req.find("id");
135  auto method_it = req.find("method");
136  auto params_it = req.find("params");
137 
138  if (method_it == req.end()) {
139  PACKIO_ERROR("missing method field");
140  return std::nullopt;
141  }
142  if (!method_it->value().is_string()) {
143  PACKIO_ERROR("method field is not a string");
144  return std::nullopt;
145  }
146 
147  std::optional<request> parsed{std::in_place};
148  parsed->method = std::string{
149  method_it->value().get_string().data(),
150  method_it->value().get_string().size(),
151  };
152  if (params_it == req.end() || params_it->value().is_null()) {
153  parsed->args = boost::json::array{};
154  }
155  else if (!params_it->value().is_array() && !params_it->value().is_object()) {
156  PACKIO_ERROR("non-structured arguments are not supported");
157  return std::nullopt;
158  }
159  else {
160  parsed->args = std::move(params_it->value());
161  }
162 
163  if (id_it == req.end() || id_it->value().is_null()) {
164  parsed->type = call_type::notification;
165  }
166  else {
167  parsed->type = call_type::request;
168  parsed->id = std::move(id_it->value());
169  }
170  return parsed;
171  }
172 
173  std::vector<char> buffer_;
174  std::queue<boost::json::value> parsed_;
175  std::unique_ptr<boost::json::stream_parser> parser_;
176 };

331  template <typename T, typename NamesContainer>
332  static std::optional<T> extract_args(
333  boost::json::value&& args,
334  const NamesContainer& names)
335  {
336  try {
337  if (args.is_array()) {
338  if (args.get_array().size() != std::tuple_size_v<T>) {
339  // keep this check otherwise the converter
340  // may silently drop arguments
341  PACKIO_WARN(
342  "cannot convert args: wrong number of arguments");
343  return std::nullopt;
344  }
345 
346  return convert_positional_args<T>(args.get_array());
347  }
348  else if (args.is_object()) {
349  return convert_named_args<T>(args.get_object(), names);
350  }
351  else {
352  PACKIO_ERROR("arguments are not a structured type");
353  return std::nullopt;
354  }
355  }
356  catch (const std::exception& exc) {
357  PACKIO_WARN("cannot convert args: {}", exc.what());
358  (void)exc;
359  return std::nullopt;
360  }
361  }
