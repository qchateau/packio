219  template <typename Buffer, typename WriteHandler>
220  void async_send(std::unique_ptr<Buffer>&& buffer_ptr, WriteHandler&& handler)
221  {
222  wstrand_.push([self = shared_from_this(),
223  buffer_ptr = std::move(buffer_ptr),
224  handler = std::forward<WriteHandler>(handler)]() mutable {
225  assert(self->strand_.running_in_this_thread());
226  internal::set_no_delay(self->socket_);
227 
228  auto buf = rpc_type::buffer(*buffer_ptr);
229  net::async_write(
230  self->socket_,

233  self->strand_,
234  [self,
235  buffer_ptr = std::move(buffer_ptr),

244  void async_read(parser_type&& parser)
245  {
246  parser.reserve_buffer(buffer_reserve_size_);
247  auto buffer = net::buffer(parser.buffer(), parser.buffer_capacity());
248 
249  assert(strand_.running_in_this_thread());
250  reading_ = true;
251  PACKIO_TRACE("reading ... {} call(s) pending", pending_.size());
252  socket_.async_read_some(
253  buffer,
254  internal::bind_executor(
255  strand_,
256  [this, self = shared_from_this(), parser = std::move(parser)](
257 
258  error_code ec, size_t length) mutable {
259  // stop if there is an error or there is no more pending calls
260  assert(self->strand_.running_in_this_thread());
261 
262  if (ec) {
263  PACKIO_WARN("read error: {}", ec.message());
264  self->reading_ = false;
265  if (ec != net::error::operation_aborted)
266  self->close();
267  return;
268  }
269 
270  PACKIO_TRACE("read: {}", length);
271  parser.buffer_consumed(length);
272 
273  while (true) {
274  auto response = parser.get_response();
275  if (!response) {
276  PACKIO_INFO("stop reading: {}", response.error());
277  break;
278  }
279  self->async_call_handler(std::move(*response));
280  }
281 
282  if (self->pending_.empty()) {
283  PACKIO_TRACE("done reading, no more pending calls");
284  self->reading_ = false;
285  return;
286  }
287 
288  self->async_read(std::move(parser));
289  }));
290  }

298  void async_call_handler(id_type id, error_code ec, response_type&& response)
299  {
300  net::dispatch(
301  strand_,
302  [ec,
303  id,
304  self = shared_from_this(),
305  response = std::move(response)]() mutable {
306  PACKIO_DEBUG(
307  "calling handler for id: {}", rpc_type::format_id(id));
308 
309  assert(self->strand_.running_in_this_thread());
310  auto it = self->pending_.find(id);
311  if (it == self->pending_.end()) {
312  PACKIO_WARN("unexisting id: {}", rpc_type::format_id(id));
313  return;
314  }
315 
316  auto handler = std::move(it->second);
317  self->pending_.erase(it);
318 
319  // handle the response asynchronously (post)
320  // to schedule the next read immediately
321  // this will allow parallel response handling
322  // in multi-threaded environments
323  net::post(
324  self->socket_.get_executor(),
325  [ec,
326  handler = std::move(handler),
327  response = std::move(response)]() mutable {
328  handler(ec, std::move(response));
329  });
330  });
331  }

395  template <typename CallHandler, typename ArgsTuple>
396  void operator()(
397  CallHandler&& handler,
398  std::string_view name,
399  ArgsTuple&& args,
400  std::optional<std::reference_wrapper<id_type>> opt_call_id) const
401  {
402  PACKIO_STATIC_ASSERT_TTRAIT(CallHandler, rpc_type);
403  PACKIO_DEBUG("async_call: {}", name);
404 
405  id_type call_id = self_->id_.fetch_add(1, std::memory_order_acq_rel);
406  if (opt_call_id) {
407  opt_call_id->get() = call_id;
408  }
409 
410  auto packer_buf = internal::to_unique_ptr(std::apply(
411  [&name, &call_id](auto&&... args) {
412  return rpc_type::serialize_request(
413  call_id, name, std::forward<decltype(args)>(args)...);
414  },
415  std::forward<ArgsTuple>(args)));
416 
417  net::dispatch(
418  self_->strand_,
419  [self = self_->shared_from_this(),
420  call_id,
421  handler = std::forward<CallHandler>(handler),
422  packer_buf = std::move(packer_buf)]() mutable {
423  // we must emplace the id and handler before sending data
424  // otherwise we might drop a fast response
425  assert(self->strand_.running_in_this_thread());
426  self->pending_.try_emplace(call_id, std::move(handler));
427 
428  // if we are not reading, start the read operation
429  if (!self->reading_) {
430  PACKIO_DEBUG("start reading");
431  self->async_read(parser_type{});
432  }
433 
434  // send the request buffer
435  self->async_send(
436  std::move(packer_buf),
437  [self = std::move(self), call_id](
438  error_code ec, std::size_t length) mutable {
439  if (ec) {
440  PACKIO_WARN("write error: {}", ec.message());
441  if (ec != net::error::operation_aborted)
442  self->close();
443  return;
444  }
445 
446  PACKIO_TRACE("write: {}", length);
447  (void)length;
448  });
449  });
450  }
