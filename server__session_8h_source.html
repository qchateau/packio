71  void async_read(parser_type&& parser)
72  {
73  // abort R/W on error
74  if (!socket_.is_open()) {
75  return;
76  }
77 
78  parser.reserve_buffer(buffer_reserve_size_);
79  auto buffer = net::buffer(parser.buffer(), parser.buffer_capacity());
80  socket_.async_read_some(
81  buffer,
82  [self = shared_from_this(), parser = std::move(parser)](
83  error_code ec, size_t length) mutable {
84  if (ec) {
85  PACKIO_WARN("read error: {}", ec.message());
86  self->close_connection();
87  return;
88  }
89 
90  PACKIO_TRACE("read: {}", length);
91  parser.buffer_consumed(length);
92 
93  while (auto request = parser.get_request()) {
94  // handle the call asynchronously (post)
95  // to schedule the next read immediately
96  // this will allow parallel call handling
97  // in multi-threaded environments
98  net::post(
99  self->get_executor(),
100  [self, request = std::move(*request)]() mutable {
101  self->async_handle_request(std::move(request));
102  });
103  }
104 
105  self->async_read(std::move(parser));
106  });
107  }
108 
109  void async_handle_request(request_type&& request)
110  {
111  completion_handler<Rpc> handler(
112  request.id,
113  [type = request.type, id = request.id, self = shared_from_this()](
114  auto&& response_buffer) {
115  if (type == call_type::request) {
116  PACKIO_TRACE("result (id={})", Rpc::format_id(id));
117  (void)id;
118  self->async_send_response(std::move(response_buffer));
119  }
120  });
121 
122  const auto function = dispatcher_ptr_->get(request.method);
123  if (function) {
124  PACKIO_TRACE(
125  "call: {} (id={})", request.method, Rpc::format_id(request.id));
126  (*function)(std::move(handler), std::move(request.args));
127  }
128  else {
129  PACKIO_DEBUG("unknown function {}", request.method);
130  handler.set_error("Unknown function");
131  }
132  }

134  template <typename Buffer>
135  void async_send_response(Buffer&& response_buffer)
136  {
137  // abort R/W on error
138  if (!socket_.is_open()) {
139  return;
140  }
141 
142  auto message_ptr = internal::to_unique_ptr(std::move(response_buffer));
143 
144  wstrand_.push([this,
145  self = shared_from_this(),
146  message_ptr = std::move(message_ptr)]() mutable {
147  auto buf = Rpc::buffer(*message_ptr);
148  net::async_write(
149  socket_,
150  buf,
151  [self = std::move(self), message_ptr = std::move(message_ptr)](
152  error_code ec, size_t length) {
153  self->wstrand_.next();
154 
155  if (ec) {
156  PACKIO_WARN("write error: {}", ec.message());
157  self->close_connection();
158  return;
159  }
160 
161  PACKIO_TRACE("write: {}", length);
162  (void)length;
163  });
164  });
165  }
166 
167  void close_connection()
168  {
169  error_code ec;
170  socket_.close(ec);
171  if (ec) {
172  PACKIO_WARN("close error: {}", ec.message());
173  }
174  }
