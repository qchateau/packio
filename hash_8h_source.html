14 inline std::size_t combine(std::size_t seed, std::size_t h) noexcept
15 {
16  seed ^= h + 0x9e3779b9 + (seed << 6U) + (seed >> 2U);
17  return seed;
18 }

20 inline std::size_t hash(const boost::json::value& v)
21 {
22  std::size_t seed = 0;
23  switch (v.kind()) {
24  case boost::json::kind::null:
25  // Don't use std::hash<std::nullptr_t>, old gcc/clang do not support it
26  return combine(seed, std::hash<void*>()(nullptr));
27  case boost::json::kind::bool_:
28  return combine(seed, std::hash<bool>{}(v.get_bool()));
29  case boost::json::kind::uint64:
30  return combine(seed, std::hash<uint64_t>{}(v.get_uint64()));
31  case boost::json::kind::int64:
32  return combine(seed, std::hash<int64_t>{}(v.get_int64()));
33  case boost::json::kind::double_:
34  return combine(seed, std::hash<double>{}(v.get_double()));
35  case boost::json::kind::string:
36  return combine(
37  seed,
38  std::hash<std::string_view>{}(std::string_view{
39  v.get_string().data(), v.get_string().size()}));
40  case boost::json::kind::array: {
41  seed = combine(seed, v.get_array().size());
42  for (const auto& element : v.get_array()) {
43  seed = combine(seed, hash(element));
44  }
45  return seed;
46  }
47  case boost::json::kind::object: {
48  seed = combine(seed, v.get_object().size());
49  for (const auto& element : v.get_object()) {
50  std::string_view key{element.key().data(), element.key().size()};
51  seed = combine(seed, std::hash<std::string_view>{}(key));
52  seed = combine(seed, hash(element.value()));
53  }
54  return seed;
55  }
56  }
57 }
